## Install

```bash
npm install --save buttonarts
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'buttonarts'
import 'buttonarts/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
